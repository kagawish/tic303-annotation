    package eg.ufe.kagawish.entreprise3;

import java.util.ArrayList;
import java.util.Comparator;
import java.util.Iterator;
import java.util.List;

public class TestEntreprise {

	public static void main(String[] args) {
		Entreprise e1 = new Entreprise("IBM");
		Entreprise e2 = new Entreprise("Sun");
		e1.engager(new Employe("Dupond", 800));
		e1.engager(new Employe("Poiret", 900));
		e2.engager(new Employe("Ravier", 750));
		e2.engager(new Employe("Burot", 1200));
		e2.engager(new Employe("Terrice", 1000));
		e2.engager(new Employe("Pernaut", 1150));

		double somme = 0, somme2 = 0;

		Iterator<Employe> it = e1.iterEmployes();
		while (it.hasNext()) {
			Employe e = it.next();
			somme += e.getSalaire();
		}

		Iterator<Employe> it2 = e2.iterator();
		while (it2.hasNext()) {
			Employe e = it2.next();
			somme2 += e.getSalaire();
		}

		System.out.println(somme);
		System.out.println(somme2);

		double somme3 = 0;
		for (Employe e : e1) {
			somme3 += e.getSalaire();
		}
		System.out.println(somme3);

		List<Employe> l = new ArrayList<Employe>();
		for (Employe e : e2) {
			l.add(e);
		}
		for (Employe e : e1) {
			l.add(e);
		}

		System.out.println(l);

		l.sort(Comparator.comparing(Employe::getSalaire).reversed());

		System.out.println(l);
	}
}
