    package eg.ufe.kagawish.etageresexception;

import java.util.ArrayList;

public class Librairie {

	private static final int TAILLE_ETAGERES = 2;
	private ArrayList<Etagere> etageres = new ArrayList<Etagere>();

	public Librairie() {
		etageres.add(new Etagere(TAILLE_ETAGERES));
	}

	public void ajouterLivre(Livre livre) {
		try {
			etageres.get(etageres.size() - 1).ajouter(livre);
		} catch (EtagerePleineException ex) {
			etageres.add(new Etagere(TAILLE_ETAGERES));
			this.ajouterLivre(livre);
		}
	}

	public void afficheLivres() {
		for (Etagere etagere : etageres) {
			System.out.println("Etagere: {");
			for (int i = 0; i < etagere.getNbLivres(); i++) {
				System.out.print("Livre num√©ro " + (i + 1) + " : ");
				System.out.println(etagere.getLivre(i));
			}
			System.out.println("}");
		}
	}

	public static void main(String[] args) {
		Livre l1 = new Livre("a1", "t1");
		Livre l2 = new Livre("a2", "t2");
		Livre l3 = new Livre("a3", "t3");
		Livre l4 = new Livre("a4", "t4");
		Livre l5 = new Livre("a5", "t5");
		Livre l6 = new Livre("a6", "t6");
		Livre l7 = new Livre("a7", "t7");
		Librairie librairie = new Librairie();
		librairie.ajouterLivre(l1);
		librairie.ajouterLivre(l2);
		librairie.ajouterLivre(l3);
		librairie.ajouterLivre(l4);
		librairie.ajouterLivre(l5);
		librairie.ajouterLivre(l6);
		librairie.ajouterLivre(l7);
		librairie.afficheLivres();
	}

}
