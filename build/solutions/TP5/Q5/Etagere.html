    package eg.ufe.kagawish.iterateur;

import java.util.Arrays;
import java.util.ConcurrentModificationException;
import java.util.Iterator;
import java.util.NoSuchElementException;

public class Etagere {

	private Livre[] livres;
	private int nbLivres = 0;
	private int nombreDeModifs = 0;

	public Etagere(int nb) {
		livres = new Livre[nb];
	}

	public int getNbMaxLivres() {
		return livres.length;
	}

	public int getNbLivres() {
		return nbLivres;
	}

	public void ajouter(Livre l) {
		if (nbLivres < livres.length) {
			livres[nbLivres++] = l;
			nombreDeModifs++;
		} else {
			System.err.println("Etagere pleine");
		}
	}

	public Livre getLivre(int i) {
		return livres[i];
	}

	public String toString() {
		return "Etagere [nbLivres=" + nbLivres + ", livres=" + Arrays.toString(livres) + "]";
	}

	public Iterator<Livre> iterator() {
		return new IteratorLivre();
	}

	class IteratorLivre implements Iterator<Livre> {

		private int indexIter = 0;
		private int modifs = nombreDeModifs;

		@Override
		public boolean hasNext() {
			checkModifs();

			if (indexIter < livres.length) {
				return true;
			} else {
				return false;
			}
		}

		@Override
		public Livre next() {
			checkModifs();
			checkNotNull();

			Livre l = livres[indexIter];
			indexIter++;

			return l;
		}

		@Override
		public void remove() {
			checkModifs();
			checkNotNull();

			if (indexIter == 0) {
				throw new IllegalStateException();
			}

			livres[indexIter] = livres[nbLivres - 1];
			livres[nbLivres] = null;
			nbLivres--;
		}

		private void checkNotNull() {
			if (livres[indexIter] == null) {
				throw new NoSuchElementException();
			}
		}

		private void checkModifs() {
			if (this.modifs != nombreDeModifs) {
				throw new ConcurrentModificationException();
			}
		}
	}
}
