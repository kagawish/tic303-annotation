    package eg.ufe.kagawish.cc1.ex1;

public class TestCroissant {
    public static void main(String[] args) {
        int[] nombres1 = {7, 8, 765, 987};
        
        System.out.println(croissant(nombres1));
        
        int[] nombres2 = {1, 8, 765, 65, 987};
        
        System.out.println(croissant(nombres2));
    }

    public static boolean croissant(int[] chiffres) {
        for(int i = 0; i < chiffres.length - 1; i++) {
            if (chiffres[i] > chiffres[i + 1]) {
                return false;
            }
        }
        
        return true;
    }
}
