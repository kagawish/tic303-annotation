    package eg.ufe.kagawish.cc1.ex1;

public class Etudiant {

    private String nom;
    private int age;

    public Etudiant(String nom, int age) {
        this.nom = nom;
        this.age = age;
    }

    public String getNom() {
        return nom;
    }

    public int getAge() {
        return age;
    }
    
    public static boolean agesCroissants(Etudiant[] e) {
        for(int i = 0; i < e.length - 1; i++) {
            if (e[i] == null) {
                break;
            }
            if (e[i].getAge() > e[i+1].getAge()) {
                return false;
            }
        }
        return true;
    }
}
