    package eg.ufe.kagawish.cc1.ex1;

public class Test {
    public static void main(String[] args) {
        Etudiant[] etudiants = {
            new Etudiant("Mohamed", 18),
            new Etudiant("Meryem", 22)
        };
        
        System.out.println(Etudiant.agesCroissants(etudiants));
        
        etudiants = ajouterEtudiant(etudiants, new Etudiant("Hossam", 21));
        
        System.out.print(Etudiant.agesCroissants(etudiants));
    }
    
    private static Etudiant[] ajouterEtudiant(Etudiant[] etudiants, Etudiant e) {
        Etudiant[] newEtudiants = new Etudiant[etudiants.length + 1];
        
        for(int i = 0; i < etudiants.length; i++) {
            newEtudiants[i] = etudiants[i];
        }
        newEtudiants[newEtudiants.length - 1] = e;
        
        return newEtudiants;
    }
}
